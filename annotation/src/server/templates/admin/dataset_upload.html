    Format:

  ]
}
              
              
{
  "data": [
      {
            "text": "Louis Armstrong, the great trumpet player, lived in Corona.",
            "annotations": [
                {
                      "label": "LOC",
                      "start_offset": 52,
                      "end_offset": 58
                }
                ...
            ],
            "metadata": {"foo" : "bar"}
      }
      ...
  ]
}
